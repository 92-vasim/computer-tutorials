    Logical operations:
        a. If formula 
        syntax: =IF(logical_test,[value_if_true],[value_if_false])

        b. Pass or Fail example 
        syntax: =IF(C5>=70,"Pass","Fail")

        c. Returns formula 
        syntax: =IF(A1<100,A1*5%,A1*7%)

        d. Nested If 
        syntax: =IF(C6<70,"F",IF(C6<75,"D",IF(C6<85,"C",IF(C6<95,"B","A"))))
